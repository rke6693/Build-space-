whale_scanner.py
import asyncio
from omnisight import OmniSight

async def scan_whales():
    async with OmniSight(api_key="sk-...") as client:
        # Stream real-time whale alerts
        async for alert in client.stream_whale_alerts():
            if alert.usd_value > 100_000:
                print(f"MEGA WHALE: {alert.side} "
                      f"${alert.usd_value:,.0f} on "
                      f"{alert.market_title}")
                print(f"  Wallet: {alert.wallet_address}")
                print(f"  Tags: {alert.tags}")

asyncio.run(scan_whales())
